確診趨勢 疫苗分析 AI 預測

Step 1: 載入疫情數據

import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

# 載入確診數據
cases_df = pd.read_csv('data/covid_cases.csv')
cases_df['Date'] = pd.to_datetime(cases_df['Date'])

# 載入疫苗數據
vaccine_df = pd.read_csv('data/covid_vaccines.csv')
vaccine_df['Date'] = pd.to_datetime(vaccine_df['Date'])

# 計算7日移動平均
cases_df['Cases_7MA'] = cases_df['New_Cases'].rolling(window=7).mean()

print(f"時間範圍: {cases_df['Date'].min()} ~ {cases_df['Date'].max()}")
print(f"總確診數: {cases_df['New_Cases'].sum():,}")
print(f"疫苗覆蓋率: {vaccine_df['Coverage_Rate'].iloc[-1]:.1f}%")

Step 2: 確診趨勢視覺化

fig, axes = plt.subplots(2, 1, figsize=(14, 10))

# 每日確診與移動平均
axes[0].bar(cases_df['Date'], cases_df['New_Cases'],
            alpha=0.4, color='#55a3ff', label='每日確診')
axes[0].plot(cases_df['Date'], cases_df['Cases_7MA'],
             color='#003d82', linewidth=2, label='7日移動平均')
axes[0].set_title('COVID-19 每日確診趨勢', fontsize=16)
axes[0].legend()
axes[0].set_ylabel('確診人數')

# 累計確診
cases_df['Cumulative'] = cases_df['New_Cases'].cumsum()
axes[1].fill_between(cases_df['Date'], cases_df['Cumulative'],
                     alpha=0.5, color='#55a3ff')
axes[1].plot(cases_df['Date'], cases_df['Cumulative'], color='#003d82')
axes[1].set_title('累計確診人數', fontsize=16)
axes[1].set_ylabel('累計人數')

plt.tight_layout()
plt.savefig('output/確診趨勢.png', dpi=300)

Step 4: AI 疫情趨勢分析

from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate

llm = ChatOpenAI(model="gpt-3.5-turbo")

# 準備統計摘要
recent_cases = cases_df.tail(30)
stats = {
    "avg_daily": recent_cases['New_Cases'].mean(),
    "max_daily": recent_cases['New_Cases'].max(),
    "trend": "上升" if recent_cases['Cases_7MA'].iloc[-1] > recent_cases['Cases_7MA'].iloc[0] else "下降",
    "vaccine_rate": vaccine_df['Coverage_Rate'].iloc[-1]
}

analysis_prompt = ChatPromptTemplate.from_messages([
    ("system", "你是流行病學專家，請用繁體中文提供專業分析"),
    ("human", """根據近30天疫情數據：
- 日均確診：{avg_daily:.0f} 人
- 單日最高：{max_daily:.0f} 人
- 趨勢方向：{trend}
- 疫苗覆蓋率：{vaccine_rate:.1f}%

請分析：
1. 當前疫情階段評估
2. 與疫苗覆蓋率的關聯
3. 未來趨勢預測
4. 防疫建議""")
])

chain = analysis_prompt | llm
response = chain.invoke(stats)
print(response.content)

Step 5: 縣市疫情熱力圖

import seaborn as sns

# 載入各縣市數據
city_cases = pd.read_csv('data/covid_by_city.csv')

# 建立月份x城市熱力圖
pivot = city_cases.pivot_table(values='Cases', index='Month', columns='City')

Step 6: AI 自動產生防疫報告

from langchain_core.output_parsers import StrOutputParser

def generate_weekly_report(cases_df, vaccine_df):
    """自動產生週報"""

    # 本週數據
    this_week = cases_df.tail(7)
    last_week = cases_df.tail(14).head(7)

    weekly_stats = {
        "this_week_total": this_week['New_Cases'].sum(),
        "last_week_total": last_week['New_Cases'].sum(),
        "change_rate": (this_week['New_Cases'].sum() - last_week['New_Cases'].sum()) / last_week['New_Cases'].sum() * 100,
        "vaccine_rate": vaccine_df['Coverage_Rate'].iloc[-1]
    }

    report_prompt = ChatPromptTemplate.from_template(
        """產生一份簡潔的 COVID-19 週報摘要：

本週確診：{this_week_total:,} 人
上週確診：{last_week_total:,} 人
週變化率：{change_rate:+.1f}%
疫苗覆蓋率：{vaccine_rate:.1f}%

請包含：摘要、風險評估、建議措施（各1-2句話）"""
    )

    chain = report_prompt | llm | StrOutputParser()
    return chain.invoke(weekly_stats)

report = generate_weekly_report(cases_df, vaccine_df)
print(report)
