Step 1: 載入股市數據

import pandas as pd
import matplotlib.pyplot as plt
import mplfinance as mpf

# 載入股票數據
df = pd.read_csv('data/tw_stock_2330.csv')
df['Date'] = pd.to_datetime(df['Date'])
df.set_index('Date', inplace=True)

# 計算技術指標
df['MA5'] = df['Close'].rolling(window=5).mean()
df['MA20'] = df['Close'].rolling(window=20).mean()
df['MA60'] = df['Close'].rolling(window=60).mean()

# 計算 RSI
delta = df['Close'].diff()
gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
df['RSI'] = 100 - (100 / (1 + gain / loss))

print(f"股票：台積電 (2330)")
print(f"時間範圍: {df.index.min()} ~ {df.index.max()}")
print(f"最新收盤價: {df['Close'].iloc[-1]}")

Step 2: K線圖繪製

import mplfinance as mpf

# 準備 K 線數據（最近60個交易日）
kline_data = df.tail(60).copy()

# 設定均線
add_plots = [
    mpf.make_addplot(kline_data['MA5'], color='blue', width=1),
    mpf.make_addplot(kline_data['MA20'], color='orange', width=1),
    mpf.make_addplot(kline_data['RSI'], panel=1, color='purple', ylabel='RSI')
]

# 繪製 K 線圖
mpf.plot(kline_data, type='candle', style='yahoo',
         title='台積電 (2330) K線圖',
         ylabel='股價',
         volume=True,
         addplot=add_plots,
         figsize=(14, 10),
         savefig='output/K線圖.png')
K線圖結合均線與RSI指標，提供完整技術分析視角

Step 3: 技術指標視覺化

fig, axes = plt.subplots(3, 1, figsize=(14, 12), sharex=True)

Step 4: AI 技術面分析

from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate

llm = ChatOpenAI(model="gpt-3.5-turbo")

# 準備技術指標數據
latest = df.iloc[-1]
stats = {
    "close": latest['Close'],
    "ma5": latest['MA5'],
    "ma20": latest['MA20'],
    "ma60": latest['MA60'],
    "rsi": latest['RSI'],
    "volume": latest['Volume'],
    "change": (latest['Close'] - df.iloc[-2]['Close']) / df.iloc[-2]['Close'] * 100
}

analysis_prompt = ChatPromptTemplate.from_messages([
    ("system", "你是專業股票分析師，請用繁體中文提供技術面分析。注意：這僅供教學參考，不構成投資建議。"),
    ("human", """台積電 (2330) 最新數據：
- 收盤價：{close:.0f} 元
- 5日均線：{ma5:.0f} 元
- 20日均線：{ma20:.0f} 元
- 60日均線：{ma60:.0f} 元
- RSI：{rsi:.1f}
- 日漲跌：{change:+.2f}%

請分析：
1. 均線排列型態
2. RSI 超買超賣判斷
3. 短期技術面觀察
4. 關鍵支撐/壓力位""")
])

chain = analysis_prompt | llm
response = chain.invoke(stats)
print(response.content)

Step 5: AI 產業比較分析

from langchain_core.output_parsers import JsonOutputParser

# 載入多檔股票
stocks = {
    '2330': '台積電',
    '2317': '鴻海',
    '2454': '聯發科'
}

comparison_data = []
for code, name in stocks.items():
    stock_df = pd.read_csv(f'data/tw_stock_{code}.csv')
    stock_df['Date'] = pd.to_datetime(stock_df['Date'])
    latest = stock_df.iloc[-1]
    comparison_data.append({
        "code": code,
        "name": name,
        "price": latest['Close'],
        "change_30d": (latest['Close'] - stock_df.iloc[-22]['Close']) / stock_df.iloc[-22]['Close'] * 100
    })

compare_prompt = ChatPromptTemplate.from_messages([
    ("system", """你是半導體產業分析師。比較股票並輸出JSON：
{{"ranking": ["股票代碼"], "analysis": "分析說明", "top_pick": "首選股票"}}"""),
    ("human", "比較以下股票近月表現：\\n{data}")
])

parser = JsonOutputParser()
compare_chain = compare_prompt | llm | parser

result = compare_chain.invoke({"data": str(comparison_data)})
print(f"排名: {result['ranking']}")
print(f"分析: {result['analysis']}")

Step 6: AI 投資策略建議

from langchain_core.pydantic_v1 import BaseModel, Field
from typing import List

class InvestmentAdvice(BaseModel):
    action: str = Field(description="建議動作：買入/賣出/觀望")
    confidence: int = Field(description="信心度 0-100")
    reasons: List[str] = Field(description="理由列表")
    risk_level: str = Field(description="風險等級：低/中/高")
    target_price: float = Field(description="目標價位")

from langchain.output_parsers import PydanticOutputParser

advice_parser = PydanticOutputParser(pydantic_object=InvestmentAdvice)

advice_prompt = ChatPromptTemplate.from_messages([
    ("system", """你是投資顧問。根據技術指標提供建議（僅供教學參考）。
{format_instructions}"""),
    ("human", """股票：{stock}
收盤價：{price}
RSI：{rsi}
均線位置：股價{"高於" if {above_ma} else "低於"}20日均線""")
])

advice_chain = advice_prompt | llm | advice_parser

result = advice_chain.invoke({
    "stock": "台積電 (2330)",
    "price": latest['Close'],
    "rsi": latest['RSI'],
    "above_ma": latest['Close'] > latest['MA20'],
    "format_instructions": advice_parser.get_format_instructions()
})

print(f"建議：{result.action}")
print(f"信心度：{result.confidence}%")
print(f"風險：{result.risk_level}")
print(f"理由：{', '.join(result.reasons)}")
