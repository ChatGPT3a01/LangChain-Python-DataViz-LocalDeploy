Step 1: 載入社群數據

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# 載入社群媒體數據
df = pd.read_csv('data/social_media_posts.csv')
df['Post_Date'] = pd.to_datetime(df['Post_Date'])
df['Day_of_Week'] = df['Post_Date'].dt.day_name()
df['Hour'] = df['Post_Date'].dt.hour

# 計算互動率
df['Engagement_Rate'] = (df['Likes'] + df['Comments'] + df['Shares']) / df['Followers'] * 100

print(f"貼文數量: {len(df):,}")
print(f"內容類型: {df['Content_Type'].unique()}")
print(f"平均互動率: {df['Engagement_Rate'].mean():.2f}%")

Step 2: 互動數據視覺化

fig, axes = plt.subplots(2, 2, figsize=(14, 10))

# 各類型貼文互動
type_engagement = df.groupby('Content_Type')['Engagement_Rate'].mean().sort_values(ascending=False)
axes[0,0].bar(type_engagement.index, type_engagement.values, color='#e84393')
axes[0,0].set_title('各類型貼文平均互動率')
axes[0,0].tick_params(axis='x', rotation=45)

# 每日互動分布
daily_engagement = df.groupby('Day_of_Week')['Engagement_Rate'].mean()
day_order = ['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday', 'Sunday']
axes[0,1].plot([daily_engagement.get(d, 0) for d in day_order], 'o-', color='#e84393')
axes[0,1].set_xticklabels(['一', '二', '三', '四', '五', '六', '日'])
axes[0,1].set_title('每週互動趨勢')

# 時段分析
hourly = df.groupby('Hour')['Engagement_Rate'].mean()
axes[1,0].fill_between(hourly.index, hourly.values, alpha=0.5, color='#fd79a8')
axes[1,0].plot(hourly.index, hourly.values, color='#e84393')
axes[1,0].set_title('24小時互動分布')

# 散佈圖
axes[1,1].scatter(df['Likes'], df['Comments'], alpha=0.5, c='#e84393')
axes[1,1].set_xlabel('按讚數')
axes[1,1].set_ylabel('留言數')
axes[1,1].set_title('按讚 vs 留言關係')

plt.tight_layout()
plt.savefig('output/互動分析.png', dpi=300)

Step 3: AI 情感分析

from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import JsonOutputParser

llm = ChatOpenAI(model="gpt-3.5-turbo")

sentiment_prompt = ChatPromptTemplate.from_messages([
    ("system", """你是情感分析專家。分析貼文情感並輸出JSON：
{{"sentiment": "正面/中性/負面", "score": 0-100, "keywords": ["關鍵詞"]}}"""),
    ("human", "分析這則貼文：{post}")
])

parser = JsonOutputParser()
sentiment_chain = sentiment_prompt | llm | parser

# 分析貼文
sample_posts = df['Content'].head(10)
results = []

for post in sample_posts:
    result = sentiment_chain.invoke({"post": post})
    results.append(result)
    print(f"情感: {result['sentiment']}, 分數: {result['score']}")

Step 4: AI 留言摘要分析

# 載入留言數據
comments_df = pd.read_csv('data/post_comments.csv')

# 彙整特定貼文的留言
post_id = df.iloc[0]['Post_ID']
post_comments = comments_df[comments_df['Post_ID'] == post_id]['Comment'].tolist()

comment_prompt = ChatPromptTemplate.from_messages([
    ("system", "你是社群分析專家，請用繁體中文分析留言"),
    ("human", """以下是一則貼文的留言：
{comments}

請分析：
1. 主要討論主題
2. 用戶情緒傾向
3. 常見問題或建議
4. 值得關注的意見""")
])

chain = comment_prompt | llm
response = chain.invoke({"comments": "\\n".join(post_comments[:20])})
print(response.content)

Step 5: AI 內容策略建議

from langchain_core.output_parsers import StrOutputParser

# 準備表現數據
performance = df.groupby('Content_Type').agg({
    'Engagement_Rate': 'mean',
    'Likes': 'mean',
    'Shares': 'mean'
}).round(2)

strategy_prompt = ChatPromptTemplate.from_messages([
    ("system", "你是社群行銷專家，請提供具體可執行的建議"),
    ("human", """根據以下各類型貼文表現數據：
{performance}

目標受眾: {audience}
品牌調性: {tone}

請提供：
1. 最佳發文類型建議
2. 內容主題方向
3. 發文時間建議
4. 提升互動的具體方法""")
])

chain = strategy_prompt | llm | StrOutputParser()
strategy = chain.invoke({
    "performance": performance.to_string(),
    "audience": "25-35歲都會上班族",
    "tone": "專業但親切"
})
print(strategy)

Step 6: AI 文案產生器

from langchain_core.pydantic_v1 import BaseModel, Field
from typing import List

class SocialPost(BaseModel):
    main_text: str = Field(description="主要文案內容")
    hashtags: List[str] = Field(description="推薦標籤")
    call_to_action: str = Field(description="行動呼籲")
    best_time: str = Field(description="建議發布時間")

from langchain.output_parsers import PydanticOutputParser

post_parser = PydanticOutputParser(pydantic_object=SocialPost)

copywriting_prompt = ChatPromptTemplate.from_messages([
    ("system", """你是社群文案專家。為品牌產生吸引人的貼文。
{format_instructions}"""),
    ("human", "品牌：{brand}\\n產品：{product}\\n目標：{goal}")
])

copy_chain = copywriting_prompt | llm | post_parser

result = copy_chain.invoke({
    "brand": "健康生活館",
    "product": "有機蔬果箱",
    "goal": "提升訂閱數",
    "format_instructions": post_parser.get_format_instructions()
})

print(f"文案：{result.main_text}")
print(f"標籤：{' '.join(['#'+tag for tag in result.hashtags])}")
print(f"CTA：{result.call_to_action}")
